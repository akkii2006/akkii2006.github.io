const evomind = require('evomind');

// Initialize the client
const client = new evomind.Client({
    apiKey: 'your_api_key'
});

// Check a message
async function checkMessage() {
    try {
        const result = await client.analyzeMessage({
            content: 'Hello world!',
            context: 'group_chat',
            sensitivity: 'medium'
        });

        if (result.isSafe) {
            console.log('Message is safe to send');
        } else {
            console.log('Detected issues:', result.issues);
        }
    } catch (error) {
        console.error('Error:', error);
    }
}

checkMessage();
